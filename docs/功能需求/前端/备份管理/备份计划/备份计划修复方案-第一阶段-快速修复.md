### 2.2 修复2：查询last_run_time

**问题**：后端API硬编码`last_run_time`为`None`，没有从执行日志表查询

**修改方案**：在获取备份计划列表和单个备份计划时，从`backup_execution_logs`表查询最后执行时间

#### 修改文件2：`app/api/endpoints/configurations.py`

**位置**：第80-131行

**修改为**：
```python
from sqlalchemy import func

@router.get("/backup-schedules", response_model=List[Dict])
async def get_backup_schedules(
    skip: int = 0,
    limit: int = 100,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """获取备份计划列表"""
    try:
        results = db.query(BackupSchedule, Device.name).join(
            Device, BackupSchedule.device_id == Device.id
        ).offset(skip).limit(limit).all()
        
        # 批量查询每个计划的最后执行时间
        schedule_ids = [schedule.id for schedule, _ in results]
        last_executions = db.query(
            BackupExecutionLog.schedule_id,
            func.max(BackupExecutionLog.completed_at).label('last_run')
        ).filter(
            BackupExecutionLog.schedule_id.in_(schedule_ids)
        ).group_by(BackupExecutionLog.schedule_id).all()
        
        last_run_dict = {se.schedule_id: se.last_run for se in last_executions}
        
        schedules = []
        for schedule, device_name in results:
            schedule_dict = {
                'id': schedule.id,
                'device_id': schedule.device_id,
                'device_name': device_name,
                'schedule_type': schedule.schedule_type,
                'schedule_time': schedule.time,
                'schedule_day': schedule.day,
                'is_active': schedule.is_active,
                'last_run_time': last_run_dict.get(schedule.id),  # 从执行日志查询
                'created_at': schedule.created_at,
                'updated_at': schedule.updated_at
            }
            schedules.append(schedule_dict)
        
        return schedules
    except Exception as e:
        logger.error(f"获取备份计划列表失败: {str(e)}")
        raise HTTPException(status_code=500, detail=f"获取备份计划列表失败: {str(e)}")
```

@router.post("/backup-schedules", response_model=Dict)
async def create_backup_schedule(
    schedule_data: BackupScheduleCreate,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
    netmiko_service: NetmikoService = Depends(get_netmiko_service),
    git_service: GitService = Depends(get_git_service)
):
    """创建备份计划"""
    try:
        # 创建设备备份目录
        device = db.query(Device).filter(Device.id == schedule_data.device_id).first()
        if not device:
            raise HTTPException(status_code=404, detail="设备不存在")
        
        device_backup_dir = os.path.join(BACKUP_DIR, device.ip_address.replace('.', '_'))
        os.makedirs(device_backup_dir, exist_ok=True)
        
        # 创建备份计划
        schedule = BackupSchedule(

            schedule_type=schedule_data.schedule_type,

        db.refresh(schedule)

            "initial_backup": backup_result
        }

        raise
    except Exception as e:
        logger.error(f"创建备份计划失败: {str(e)}")

**修改为**：
```python
import uuid
from datetime import datetime

@router.post("/backup-schedules", response_model=Dict)
async def create_backup_schedule(
    schedule_data: BackupScheduleCreate,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
    netmiko_service: NetmikoService = Depends(get_netmiko_service),
    git_service: GitService = Depends(get_git_service)
):
    """创建备份计划"""
    execution_log = None
    try:
        # 创建设备备份目录
        device = db.query(Device).filter(Device.id == schedule_data.device_id).first()
        if not device:
            raise HTTPException(status_code=404, detail="设备不存在")
        
        device_backup_dir = os.path.join(BACKUP_DIR, device.ip_address.replace('.', '_'))
        os.makedirs(device_backup_dir, exist_ok=True)
        
        # 创建备份计划
        schedule = BackupSchedule(
            device_id=schedule_data.device_id,
            schedule_type=schedule_data.schedule_type,
            time=schedule_data.time,
            day=schedule_data.day,
            is_active=schedule_data.is_active
        )
        db.add(schedule)
        db.commit()
        db.refresh(schedule)
        
        # 立即执行一次备份
        task_id = str(uuid.uuid4())
        started_at = datetime.now()
        
        try:
            backup_result = await collect_config_from_device(
                schedule.device_id, db, netmiko_service, git_service
            )
            
            # 记录执行日志
            execution_log = BackupExecutionLog(
                task_id=task_id,
                device_id=schedule.device_id,
                schedule_id=schedule.id,
                status="success" if backup_result.get("success") else "failed",
                trigger_type="manual",
                error_message=backup_result.get("error"),
                started_at=started_at,
                completed_at=datetime.now(),
                config_id=backup_result.get("config_id")
            )
            db.add(execution_log)
            db.commit()
            
        except Exception as backup_error:
            # 备份执行失败，记录失败日志
            logger.error(f"创建备份计划时立即备份失败: {str(backup_error)}")
            execution_log = BackupExecutionLog(
                task_id=task_id,
                device_id=schedule.device_id,
                schedule_id=schedule.id,
                status="failed",
                trigger_type="manual",
                error_message=str(backup_error),
                started_at=started_at,
                completed_at=datetime.now()
            )
            db.add(execution_log)
            db.commit()
            backup_result = {"success": False, "error": str(backup_error)}
        
        return {
            "id": schedule.id,
            "message": "备份计划创建成功",
            "initial_backup": backup_result
        }
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"创建备份计划失败: {str(e)}")
        raise HTTPException(status_code=500, detail=f"创建备份计划失败: {str(e)}")
```
