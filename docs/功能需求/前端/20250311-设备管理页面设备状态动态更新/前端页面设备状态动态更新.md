## 1. 需求分析

### 1.1 原始需求

```
需求：
主页面设备管理菜单
当前页面："状态"，目前分为活跃、维护、离线、故障
在状态管理页面新增一个字段"延迟"，服务端按周期自动对对应设备进行登录端口的tcp连接测试，
若成功记录延迟时间为延迟字段，若失败记录延迟时间为999ms，并且将状态自动设置为离线
```

### 1.2 需求拆解

| 序号 | 功能点 | 描述 |
|------|--------|------|
| 1 | 新增延迟字段 | 在设备管理页面表格中新增"延迟"列，显示设备网络延迟 |
| 2 | TCP连接测试 | 服务端对设备登录端口进行TCP连接测试 |
| 3 | 周期性检测 | 按配置的周期自动执行延迟检测 |
| 4 | 延迟记录 | 连接成功记录实际延迟时间(ms)，失败记录999ms |
| 5 | 状态联动 | 连接失败时自动将设备状态设置为离线 |

### 4.1 延迟检测服务详细设计

#### 4.1.1 TCP连接测试原理

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                           TCP连接测试流程                                    │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  1. 记录开始时间 (start_time)                                               │
│       │                                                                     │
│       ▼                                                                     │
│  2. 创建TCP Socket                                                          │
│       │                                                                     │
│       ▼                                                                     │
│  3. 设置非阻塞模式 + 超时                                                   │
│       │                                                                     │
│       ▼                                                                     │
│  4. 尝试连接设备IP:端口                                                     │
│       │                                                                     │
│       ├─────────────────────────────────────────┐                          │
│       │                                         │                          │
│       ▼                                         ▼                          │
│  5a. 连接成功                              5b. 连接失败                      │
│       │                                         │                          │
│       ▼                                         ▼                          │
│  6a. 计算延迟时间                          6b. 记录错误                      │
│      latency = (end_time - start_time) * 1000    latency = 999              │
│       │                                         │                          │
│       ▼                                         ▼                          │
│  7a. 更新设备延迟                          7b. 更新设备状态为离线            │
│      device.latency = latency                    device.status = "offline"  │
│       │                                         │                          │
│       └─────────────────────────────────────────┘                          │
│                               │                                             │
│                               ▼                                             │
│                          8. 关闭Socket                                       │
│                               │                                             │
│                               ▼                                             │
│                          9. 返回结果                                         │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

#### 4.1.2 核心代码实现

```python
import asyncio
import socket
import time
from datetime import datetime
from typing import Optional, Dict, Any
from sqlalchemy.orm import Session

from app.models.models import Device
from app.config import settings


class LatencyService:
    """
    设备延迟检测服务
    """
    
    def __init__(self):
        self.timeout = getattr(settings, 'LATENCY_CHECK_TIMEOUT', 5)
        self.max_latency = 999  # 最大延迟值，表示连接失败
    
    async def check_device_latency(
        self, 
        device: Device, 
        db: Session
    ) -> Dict[str, Any]:
        """
        检测单个设备延迟
        
        Args:
            device: 设备对象
            db: 数据库会话
            
        Returns:
            检测结果字典
        """
        result = {
            "device_id": device.id,
            "hostname": device.hostname,
            "ip_address": device.ip_address,
            "port": device.login_port,
            "latency": None,
            "success": False,
            "error": None,
            "checked_at": datetime.now()
        }
        
        try:
            # 执行TCP连接测试
            latency = await self._tcp_connect_test(
                device.ip_address, 
                device.login_port
            )
            
            result["latency"] = latency
            result["success"] = latency < self.max_latency
            
            # 更新设备延迟和状态
            await self._update_device_latency(db, device, latency, result["success"])
            
        except Exception as e:
            result["latency"] = self.max_latency
            result["error"] = str(e)
            await self._update_device_latency(db, device, self.max_latency, False)
        
        return result
    
    async def _tcp_connect_test(
        self, 
        ip_address: str, 
        port: int
    ) -> int:
        """
        TCP连接测试
        
        Args:
            ip_address: IP地址
            port: 端口号
            
        Returns:
            延迟时间(ms)
        """
        start_time = time.time()
        
        try:
            # 创建socket
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            
            # 尝试连接
            sock.connect((ip_address, port))
            
            # 计算延迟
            end_time = time.time()
            latency = int((end_time - start_time) * 1000)
            
            # 关闭连接
            sock.close()
            
            return latency
            
        except socket.timeout:
            return self.max_latency
        except socket.error as e:
            return self.max_latency
        except Exception as e:
            return self.max_latency
    
    async def _update_device_latency(
        self, 
        db: Session, 
        device: Device, 
        latency: int, 
        success: bool
    ):
        """
        更新设备延迟和状态
        
        Args:
            db: 数据库会话
            device: 设备对象
            latency: 延迟时间
            success: 是否成功
        """
        device.latency = latency
        device.last_latency_check = datetime.now()
        
        # 连接失败时更新状态为离线
        if not success:
            device.status = "offline"
        
        db.commit()
        db.refresh(device)
    
    async def check_devices_batch(
        self, 
        device_ids: list, 
        db: Session
    ) -> list:
        """
        批量检测设备延迟
        
        Args:
            device_ids: 设备ID列表
            db: 数据库会话
            
        Returns:
            检测结果列表
        """
        results = []
        
        for device_id in device_ids:
            device = db.query(Device).filter(Device.id == device_id).first()
            if device:
                result = await self.check_device_latency(device, db)
                results.append(result)
        
        return results
```

### 4.2 定时调度器详细设计

```python
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.interval import IntervalTrigger
from datetime import datetime
from typing import List
from sqlalchemy.orm import Session

from app.models import get_db
from app.models.models import Device
from app.services.latency_service import LatencyService
from app.config import settings


class LatencyScheduler:
    """
    延迟检测定时调度器
    """
    
    def __init__(self):
        self.scheduler = AsyncIOScheduler()
        self.latency_service = LatencyService()
        self.enabled = getattr(settings, 'LATENCY_CHECK_ENABLED', True)
        self.interval_minutes = getattr(settings, 'LATENCY_CHECK_INTERVAL', 5)
        self.max_concurrent = getattr(settings, 'LATENCY_CHECK_MAX_CONCURRENT', 10)
        self.exclude_statuses = getattr(settings, 'LATENCY_CHECK_EXCLUDE_STATUSES', ['maintenance'])
    
    def start(self):
        """
        启动定时任务
        """
        if not self.enabled:
            return
        
        # 添加定时任务
        self.scheduler.add_job(
            self.check_all_devices,
            trigger=IntervalTrigger(minutes=self.interval_minutes),
            id='latency_check_job',
            replace_existing=True
        )
        
        self.scheduler.start()
    
    def stop(self):
        """
        停止定时任务
        """
        self.scheduler.shutdown()
    
    async def check_all_devices(self):
        """
        检测所有启用延迟检测的设备
        """
        db = next(get_db())
        
        try:
            # 查询需要检测的设备
            query = db.query(Device).filter(
                Device.latency_check_enabled == True,
                Device.status.notin_(self.exclude_statuses)
            )
            
            devices = query.all()
            
            # 批量检测
            for device in devices:
                try:
                    await self.latency_service.check_device_latency(device, db)
                except Exception as e:
                    print(f"设备 {device.hostname} 延迟检测失败: {str(e)}")
                    
        except Exception as e:
            print(f"延迟检测任务执行失败: {str(e)}")
        finally:
            db.close()
```

```javascript
// 延迟状态标签颜色
const getLatencyTagType = (latency) => {
  if (latency === null || latency === undefined) return 'info'
  if (latency >= 999) return 'danger'
  if (latency < 50) return 'success'
  if (latency < 100) return 'success'
  if (latency < 200) return 'warning'
  return 'danger'
}

// 延迟格式化
const formatLatency = (latency) => {
  if (latency === null || latency === undefined) return '未检测'
  if (latency >= 999) return '连接失败'
  return `${latency}ms`
}

// 手动延迟检测
const handleCheckLatency = async (device) => {
  try {
    device.checking = true
    const result = await deviceApi.checkLatency(device.id)
    
    if (result.success) {
      ElMessage.success(`延迟检测完成: ${formatLatency(result.data.latency)}`)
      // 更新设备数据
      device.latency = result.data.latency
      device.last_latency_check = result.data.checked_at
      if (!result.data.success) {
        device.status = 'offline'
      }
    }
  } catch (error) {
    ElMessage.error('延迟检测失败: ' + error.message)
  } finally {
    device.checking = false
  }
}
```
